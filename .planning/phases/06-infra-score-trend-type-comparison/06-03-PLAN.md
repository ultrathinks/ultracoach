Create `src/widgets/history/score-trend-chart.tsx`:

```typescript
"use client";

import type { ScoreTrendPoint } from "@/entities/analytics";
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  Tooltip,
  ResponsiveContainer,
} from "recharts";

interface ScoreTrendChartProps {
  data: ScoreTrendPoint[];
}

function formatDate(iso: string): string {
  const d = new Date(iso);
  return `${d.getMonth() + 1}/${d.getDate()}`;
}

function ScoreTrendChartInner({ data }: ScoreTrendChartProps) {
  if (data.length === 0) return null;

  if (data.length === 1) {
    return (
      <div className="rounded-xl bg-card border border-white/[0.1] p-6">
        <h3 className="font-semibold mb-4">점수 추이</h3>
        <div className="flex items-center justify-center h-48 text-muted text-sm">
          2개 이상 세션을 완료하면 추이를 확인할 수 있어요
        </div>
      </div>
    );
  }

  const chartData = data.map((d) => ({
    date: formatDate(d.createdAt),
    전달력: d.deliveryScore,
    답변력: d.contentScore,
  }));

  return (
    <div className="rounded-xl bg-card border border-white/[0.1] p-6">
      <h3 className="font-semibold mb-6">점수 추이</h3>
      <ResponsiveContainer width="100%" height={240}>
        <LineChart data={chartData}>
          <XAxis
            dataKey="date"
            stroke="var(--color-muted)"
            fontSize={12}
            tickLine={false}
            axisLine={false}
          />
          <YAxis
            domain={[0, 100]}
            stroke="var(--color-muted)"
            fontSize={12}
            tickLine={false}
            axisLine={false}
            width={30}
          />
          <Tooltip
            contentStyle={{
              backgroundColor: "var(--color-card)",
              border: "1px solid rgba(255,255,255,0.1)",
              borderRadius: "8px",
              color: "var(--color-foreground)",
              fontSize: "13px",
            }}
          />
          <Line
            type="monotone"
            dataKey="전달력"
            stroke="var(--color-indigo)"
            strokeWidth={2}
            dot={false}
            activeDot={{ r: 4, fill: "var(--color-indigo)" }}
          />
          <Line
            type="monotone"
            dataKey="답변력"
            stroke="var(--color-pink)"
            strokeWidth={2}
            dot={false}
            activeDot={{ r: 4, fill: "var(--color-pink)" }}
          />
        </LineChart>
      </ResponsiveContainer>
      <div className="flex gap-6 mt-5 text-sm text-secondary">
        <span className="flex items-center gap-2">
          <span className="w-2.5 h-2.5 rounded-full bg-indigo" />
          전달력
        </span>
        <span className="flex items-center gap-2">
          <span className="w-2.5 h-2.5 rounded-full bg-pink" />
          답변력
        </span>
      </div>
    </div>
  );
}

export { ScoreTrendChartInner };
```

The consuming component (history-dashboard.tsx, Plan 06-05) will use:

```typescript
import dynamic from "next/dynamic";

const ScoreTrendChart = dynamic(
  () => import("@/widgets/history/score-trend-chart").then((m) => ({ default: m.ScoreTrendChartInner })),
  { ssr: false },
);
```

Key implementation details:
- `dot={false}` -- no dots per CONTEXT.md decision (minimal line style)
- `activeDot` -- shows on hover only
- `domain={[0, 100]}` -- fixed Y axis for score context
- `type="monotone"` -- smooth curve
- Legend is manual HTML (not Recharts Legend) to match existing MiniChart legend styling
- Single session: no chart, just a message per CONTEXT.md decision
- Korean data keys (`전달력`, `답변력`) used directly so Tooltip labels appear in Korean without custom formatter
